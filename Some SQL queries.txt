  `email_id` VARCHAR(15) NOT NULL,
 
 `course_id` INT NOT NULL,
  PRIMARY KEY (`roll_no`),
  INDEX `course_id_idx` (`course_id` ASC) VISIBLE,
  
CONSTRAINT `course_id`
    FOREIGN KEY (`course_id`)
   
 REFERENCES `course`.`course_details` (`course_id`)
   
 ON DELETE CASCADE
    ON UPDATE CASCADE);
